    c = 15;
  }

  // Cas de test 4: Condition if avec else
  if (b != 30) then {
    d = 5;
  } else {
    d = 10;
  }
  print(d);

  // Cas de test 5: Boucle while
  e = 3;
  while (e != 0) {
    e = e - 1;
  }
  print(e);

  // Cas de test 6: Lecture et impression
  read("Enter value for f", f);
